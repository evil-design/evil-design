var internals_init = function() {
  if (game.custom.internals_init) {
    return;
  }
  const modding = game.modding;
  const internals = Object.values(modding).find(val => val && typeof val.shipDisconnected === "function");
  if (!internals) {
    modding.terminal.error("Error: modding internals object not found");
    return;
  }
  if (!internals.shipDisconnected.old) {
    function shipDisconnected({id} = {}) {
      if (modding.context.event && id) {
        var ship = game.findShip(id);
        if (ship) {
          modding.context.event({ name: "ship_disconnected", ship }, game);
        }
      }
      return shipDisconnected.old.apply(this, arguments);
    }
    shipDisconnected.old = internals.shipDisconnected;
    internals.shipDisconnected = shipDisconnected;
  }
  game.custom.internals_init = true;
};

function determineMVPS(ships) {
  if (ships.length === 0) {
    return [[],[]];
  }
  let currentMVPS = [[],[]];
  let currentMVPScores = [[0,0,0],[0,0,0]];
  for (let i = 0; i < 3; i++) {
    for (let ship of ships) {
      if (ship.score >= currentMVPScores[ship.team][0]) {
        currentMVPScores[ship.team][0] = ship.score ? ship.score : "";
        currentMVPS[ship.team][0] = ship.id;
      } else if (ship.score >= currentMVPScores[ship.team][1]) {
        currentMVPScores[ship.team][1] = ship.score ? ship.score : "";
        currentMVPS[ship.team][1] = ship.id;
      } else if (ship.score >= currentMVPScores[ship.team][2]) {
        currentMVPScores[ship.team][2] = ship.score ? ship.score : "";
        currentMVPS[ship.team][2] = ship.id;
      }
    }
  }
  return currentMVPS;
};

var counter = function(game) {
  for (let ship of game.ships) {
    if (ship.custom.initTime === true) {
      ship.custom.timeToHide--;
      if (ship.custom.timeToHide > 0) {
        ship.set({invulnerable:100})
      } else if (ship.custom.timeToHide === 0) {
          ship.custom.initTime = false;
        ship.hideInstructor();
      }
    }
    if (ship.custom.pannelHidden === true) {
      if (ship.custom.pannelHiddenIn >= 1) {
        ship.custom.pannelHiddenIn--;
        hideComponentTimer(ship, ship.custom.pannelHiddenIn);
        ship.set({idle:true,invulnerable:100})
      } else if (ship.custom.pannelHiddenIn === 0) {
        if (ship.custom.firstSet !== true) {
          random_set(ship)
          ship.custom.firstSet = true;
        }
        ship.custom.pannelHidden = false;
        ship.custom.pannelHiddenIn = 30;
        hidePannelComponents(ship);
        ship.set({idle:false,x: (Math.random() - 0.5) * game.options.map_size * 10,y: (Math.random() - 0.5) * game.options.map_size * 10})
      }
    }
  }
};

  let ship = event.ship;
  let alien = event.alien;
  let killer = event.killer;
  let collectible = event.collectible;
  var component = event.id ;
  switch (event.name) {
    case "ui_component_clicked":
      if (component === "close") {
        if (ship.custom.firstSet !== true && ship.type === 101) {
          random_set(ship);
          ship.custom.firstSet = true;
        }
        ship.custom.pannelHidden = false;
        ship.custom.pannelHiddenIn = 30;
        hidePannelComponents(ship);
        ship.custom.pannelType = 1;
        ship.set({idle:false,x: (Math.random() - 0.5) * game.options.map_size * 10,y: (Math.random() - 0.5) * game.options.map_size * 10,})
      }
      if (component === "nextShip") {
        if (ship.custom.pannelType === 1) {
          ship.custom.pannelType = 2;
          setPannels(ship);
        }
        else if (ship.custom.pannelType === 2) {
          ship.custom.pannelType = 1;
          setPannels(ship);
        }
      }
      if (component === "ship1") {
        if (ship.custom.pannelType === 1) {
          ship.set({
            type: 601, stats:66666666,shield:1000,generator:0
          });
        } else {
          ship.set({type:607, stats:66666666,shield:1000,generator:0});
        }
      }
      if (component === "ship2") {
        if (ship.custom.pannelType === 1) {
          ship.set({
            type: 602, stats:66666666,shield:1000,generator:0
          });
        } else {
          ship.set({type:608, stats:66666666,shield:1000,generator:0});
        }
      }
      if (component === "ship3") {
        if (ship.custom.pannelType === 1) {
          ship.set({
            type: 603, stats:66666666,shield:1000,generator:0
          });
        } else {
          ship.set({type:609, stats:66666666,shield:1000,generator:0});
        }
      }
      if (component === "ship4") {
        if (ship.custom.pannelType === 1) {
          ship.set({
            type: 604, stats:66666666,shield:1000,generator:0
          });
        } else {
          ship.set({type:610, stats:66666666,shield:1000,generator:0});
        }
      }
      if (component === "ship5") {
        if (ship.custom.pannelType === 1) {
          ship.set({
            type: 605, stats:66666666,shield:1000,generator:0
          });
        } else {
          ship.set({type:611, stats:66666666,shield:1000,generator:0});
        }
      }
      if (component === "ship6") {
        if (ship.custom.pannelType === 1) {
          ship.set({
            type: 606, stats:66666666,shield:1000,generator:0
          });
        } else {
          ship.set({type:612, stats:66666666,shield:1000,generator:0});
        }
      }
      break ;
    case "ship_spawned":
      if (ship != null) {
        if (ship.custom.init == true) {
          setPannels(ship);
          showComponents(ship);
          ship.set({idle:true, invulnerable:  900});
        }
        ship.custom.pannelHidden = true;
        ship.custom.pannelHiddenIN = 30;
        ship.set({crystals: 500})
      }
    break;
    case "alien_destroyed":
      if (alien !== null){
        if (killer !== null) {
          killer.custom.alienKills++;
          if (alien.code === 10 && alien.level === 3) {
            assignPointsToTeam(killer, 3)
          } else if (alien.code === 18 && alien.level === 2) {
            assignPointsToTeam(killer, 2)
          } else if (alien.code === 19 && alien.level === 1) {
            assignPointsToTeam(killer, 5)
          } else {
            assignPointsToTeam(killer, 1)
          }
          killer.set({points:killer.custom.aliensKills})
        }
      }
      break;
    case "ship_destroyed":
      if (ship !== null) {
        if (killer !== null) {
          killer.custom.frags++;
        if (killer.custom.team === factionOne) {
          team1_score++;
          Team1_ShipsKills++;
        } else if (killer.custom.team === factionTwo) {
          team2_score++;
          Team2_ShipsKills++;;
        }
        killer.set({points:killer.custom.aliensKills})
        }
        ship.custom.deaths++;
        if (ship.custom.hasWeapon === true) {
          for (let i = 0; i<game.ships.length;i++) {
            game.ships[i].instructorSays(`${name} lost its weapon, it is now lost, no one can take it again!`);

          }
          ship.custom.hasWeapon = false;
          hideBombInfosStart = true;
          bruh = null;
          hasWeapon = false;
        }
      }
      break;
    case "ship_disconnected":
      if (ship.team === 0) {
        team1_numbersP--;
      } else {
        team2_numbersP--;
      }
      break;
  }
};
